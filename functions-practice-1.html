Practice 1: Factorial Calculator

Task: Create a function that calculates the factorial of a number using a loop.

Example: Factorial of 5 (5!) = 5 × 4 × 3 × 2 × 1 = 120

Required: Use a for loop to calculate the factorial.


function calculateFactorial(n) {
    // Initialize result to 1
    let result = 1;
    // Loop from n down to 1
    for(let i = n; i >= 1; i--) {
        result *= i;  // Multiply current number
    }
    return result;
}

Practice 2: Sum Calculator (While Loop)

Task: Create a function that finds the sum of numbers from 1 to n using a while loop.

Example: Sum of numbers from 1 to 5 = 1 + 2 + 3 + 4 + 5 = 15

Required: Use a while loop for this calculation.


function calculateSum(n) {
    let sum = 0;
    let i = 1;
    while(i <= n) {
        sum += i;
        i++;
    }
    return sum;
}

Practice 3: Multiplication Table (Do...While Loop)

Task: Create a function that generates a multiplication table for a given number using a do...while loop.

Example: Table of 5: 5×1=5, 5×2=10, etc.

Required: Use a do...while loop to generate the table.


function generateMultiplicationTable(n) {
    let i = 1;
    let table = '';
    do {
        table += `${n} × ${i} = ${n * i}\n`;
        i++;
    } while(i <= 10);
    return table;
}

Practice 4: Power Calculator (For Loop)

Task: Create a function that calculates the power of a number (x^n) using a for loop.

Example: 2^3 = 2 × 2 × 2 = 8

Required: Use a for loop (don't use Math.pow()).



function calculatePower(base, exponent) {
    let result = 1;
    for(let i = 0; i < exponent; i++) {
        result *= base;
    }
    return result;
}
